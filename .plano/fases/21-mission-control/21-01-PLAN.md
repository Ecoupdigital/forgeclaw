export function updateWebhook(
  id: number,
  updates: Partial<Omit<import("./types").Webhook, "id" | "createdAt">>,

  const d = getDb();

  try {
    const fields: string[] = [];

    if (updates.url !== undefined) {
      fields.push("url = ?");
      values.push(updates.url);
    }
    if (updates.events !== undefined) {
      fields.push("events = ?");
      values.push(JSON.stringify(updates.events));
