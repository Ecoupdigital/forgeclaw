**4b. Implementar logica de filtragem por entity** — Quando `ctx.entityFilter` esta presente e nao vazio, em vez de usar o search FTS5 generico, usar `stateStore.findMemoriesByEntity()` para buscar memorias por cada tag e depois ranquear por relevancia.

Substituir o corpo do metodo prefetch APOS o `if (!this.shouldSearch(query)) return null;` (~linhas 58-109):

```typescript
    const entityFilter = ctx.entityFilter;
    const hasFilter = entityFilter && entityFilter.length > 0;

    // 1) Memory entries — filtered by entity tags or full FTS5
    let memHits: import('../types').MemorySearchHit[];
    if (hasFilter) {
      // Filtered mode: only memories tagged with any of the filter tags
      const uniqueEntries = new Map<number, import('../types').MemoryEntry>();
      for (const tag of entityFilter) {
        const entries = stateStore.findMemoriesByEntity('default', 'default', tag, 10);
        for (const entry of entries) {
          if (!uniqueEntries.has(entry.id)) {
            uniqueEntries.set(entry.id, entry);
          }
        }
      }
      // Convert to MemorySearchHit format with synthetic score (access_count based)
      memHits = Array.from(uniqueEntries.values()).map((entry) => ({
        entry,
        score: entry.accessCount + (entry.pinned ? 100 : 0),
        reason: 'entity-filter',
      }));
      // Sort by score descending, limit to 10
      memHits.sort((a, b) => b.score - a.score);
      memHits = memHits.slice(0, 10);
    } else {
      // Global mode: FTS5 search across all memories
      memHits = this.store.search(query, 5);
    }

    // 2) FTS5 over past messages — SKIP when filtered mode (agent topics
    //    shouldn't get random message history). Also skip when --resume active.
    const msgHits = (ctx.sessionId || hasFilter) ? [] : stateStore.searchMessages(query, 5);

    if (memHits.length === 0 && msgHits.length === 0) {
      return null;
    }

    const lines: string[] = [];
    const hitsReport: PrefetchResult['hits'] = [];

    if (memHits.length > 0) {
      lines.push('## memorias relacionadas');
      for (const h of memHits) {
        lines.push(`- [${h.entry.kind}] ${h.entry.content.slice(0, 280)}`);
        hitsReport.push({
          memoryId: h.entry.id,
          score: h.score,
          reason: `${hasFilter ? 'entity-filter' : 'fts'}:memory:${h.entry.kind}`,
          contentPreview: h.entry.content.slice(0, 120),
        });
        // mark as accessed (boosts ranking next time)
        stateStore.touchMemoryEntry(h.entry.id);
      }
      lines.push('');
    }

    if (msgHits.length > 0) {
      lines.push('## conversas anteriores relacionadas');
      for (const h of msgHits.slice(0, 3)) {
        const preview = h.content.replace(/\s+/g, ' ').slice(0, 200);
        const d = new Date(h.createdAt);
        const when = d.toISOString().slice(0, 16).replace('T', ' ');
        lines.push(`- [${when}] ${preview}`);
        hitsReport.push({
          memoryId: -h.messageId,
          score: -h.rank,
          reason: 'fts:message',
          contentPreview: preview.slice(0, 120),
        });
      }
    }

    return { text: lines.join('\n'), hits: hitsReport };
```
