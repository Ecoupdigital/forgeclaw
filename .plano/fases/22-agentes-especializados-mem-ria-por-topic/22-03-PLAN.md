```typescript
import { NextRequest, NextResponse } from "next/server";
import { getTopicAgent, updateTopicAgent, getTopic } from "@/lib/core";

// Auth middleware — grep the runtime/route.ts to find the exact auth pattern used.
// Copy it verbatim. It's likely something like:
// import { validateAuth } from "@/lib/auth"; or inline cookie check.

export async function GET(
  _req: NextRequest,
  { params }: { params: Promise<{ id: string }> },
) {
  const { id: idStr } = await params;
  const id = parseInt(idStr, 10);
  if (isNaN(id)) {
    return NextResponse.json({ error: "Invalid topic id" }, { status: 400 });
  }

  const topic = getTopic(id);
  if (!topic) {
    return NextResponse.json({ error: "Topic not found" }, { status: 404 });
  }

  const agent = getTopicAgent(id);
  return NextResponse.json({
    topicId: id,
    systemPrompt: agent?.systemPrompt ?? null,
    memoryMode: agent?.memoryMode ?? "global",
    memoryDomainFilter: agent?.memoryDomainFilter ?? null,
  });
}

export async function PUT(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> },
) {
  const { id: idStr } = await params;
  const id = parseInt(idStr, 10);
  if (isNaN(id)) {
    return NextResponse.json({ error: "Invalid topic id" }, { status: 400 });
  }

  const topic = getTopic(id);
  if (!topic) {
    return NextResponse.json({ error: "Topic not found" }, { status: 404 });
  }

  let body: Record<string, unknown>;
  try {
    body = await req.json();
  } catch {
    return NextResponse.json({ error: "Invalid JSON body" }, { status: 400 });
  }

  // Validate fields
  const updates: {
    systemPrompt?: string | null;
    memoryMode?: "global" | "filtered";
    memoryDomainFilter?: string[] | null;
  } = {};

  if ("systemPrompt" in body) {
    if (body.systemPrompt !== null && typeof body.systemPrompt !== "string") {
      return NextResponse.json(
        { error: "systemPrompt must be string or null" },
        { status: 400 },
      );
    }
    updates.systemPrompt = body.systemPrompt as string | null;
  }

  if ("memoryMode" in body) {
    if (body.memoryMode !== "global" && body.memoryMode !== "filtered") {
      return NextResponse.json(
        { error: "memoryMode must be 'global' or 'filtered'" },
        { status: 400 },
      );
    }
    updates.memoryMode = body.memoryMode;
  }

  if ("memoryDomainFilter" in body) {
    if (body.memoryDomainFilter !== null) {
      if (
        !Array.isArray(body.memoryDomainFilter) ||
        !body.memoryDomainFilter.every((t: unknown) => typeof t === "string")
      ) {
        return NextResponse.json(
          { error: "memoryDomainFilter must be string[] or null" },
          { status: 400 },
        );
      }
    }
    updates.memoryDomainFilter = body.memoryDomainFilter as string[] | null;
  }

  const ok = updateTopicAgent(id, updates);
  if (!ok) {
    return NextResponse.json(
      { error: "Failed to update agent config" },
      { status: 500 },
    );
  }

  // Return updated state
  const agent = getTopicAgent(id);
  return NextResponse.json({
    topicId: id,
    systemPrompt: agent?.systemPrompt ?? null,
    memoryMode: agent?.memoryMode ?? "global",
    memoryDomainFilter: agent?.memoryDomainFilter ?? null,
  });
}
```

```typescript
"use client";

import { useState, useEffect, useCallback, useRef, type KeyboardEvent } from "react";
import { Button } from "@/components/ui/button";
import { Loader2, X, Bot, Save, AlertCircle } from "lucide-react";

interface AgentConfig {
  systemPrompt: string | null;
  memoryMode: "global" | "filtered";
  memoryDomainFilter: string[] | null;
}

interface AgentConfigFormProps {
  topicId: number;
  /** Called after a successful save so parent can refresh topic data if needed. */
  onSaved?: () => void;
}

export function AgentConfigForm({ topicId, onSaved }: AgentConfigFormProps) {
  const [config, setConfig] = useState<AgentConfig>({
    systemPrompt: null,
    memoryMode: "global",
    memoryDomainFilter: null,
  });
  const [loading, setLoading] = useState(true);
  const [saving, setSaving] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [success, setSuccess] = useState(false);
  const [tagInput, setTagInput] = useState("");
  const tagInputRef = useRef<HTMLInputElement>(null);
  const successTimer = useRef<ReturnType<typeof setTimeout> | null>(null);

  // Fetch current config
  useEffect(() => {
    setLoading(true);
    setError(null);
    fetch(`/api/topics/${topicId}/agent`)
      .then(async (res) => {
        if (!res.ok) throw new Error("Failed to load agent config");
        const data = await res.json();
        setConfig({
          systemPrompt: data.systemPrompt ?? null,
          memoryMode: data.memoryMode ?? "global",
          memoryDomainFilter: data.memoryDomainFilter ?? null,
        });
      })
      .catch((err) => setError(err.message))
      .finally(() => setLoading(false));
  }, [topicId]);

  // Save config
  const handleSave = useCallback(async () => {
    setSaving(true);
    setError(null);
    setSuccess(false);
    try {
      const res = await fetch(`/api/topics/${topicId}/agent`, {
        method: "PUT",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(config),
      });
      if (!res.ok) {
        const data = await res.json().catch(() => ({ error: "Save failed" }));
        throw new Error(data.error ?? "Save failed");
      }
      setSuccess(true);
      if (successTimer.current) clearTimeout(successTimer.current);
      successTimer.current = setTimeout(() => setSuccess(false), 2000);
      onSaved?.();
    } catch (err) {
      setError(err instanceof Error ? err.message : "Save failed");
    } finally {
      setSaving(false);
    }
  }, [topicId, config, onSaved]);

  // Tag chip management
  const addTag = useCallback(
    (tag: string) => {
      const trimmed = tag.trim().toLowerCase();
      if (!trimmed) return;
      const current = config.memoryDomainFilter ?? [];
      if (current.includes(trimmed)) return;
      setConfig((prev) => ({
        ...prev,
        memoryDomainFilter: [...(prev.memoryDomainFilter ?? []), trimmed],
      }));
      setTagInput("");
    },
    [config.memoryDomainFilter],
  );

  const removeTag = useCallback((tag: string) => {
    setConfig((prev) => ({
      ...prev,
      memoryDomainFilter: (prev.memoryDomainFilter ?? []).filter((t) => t !== tag),
    }));
  }, []);

  const handleTagKeyDown = useCallback(
    (e: KeyboardEvent<HTMLInputElement>) => {
      if (e.key === "Enter" || e.key === ",") {
        e.preventDefault();
        addTag(tagInput);
      }
      if (e.key === "Backspace" && !tagInput && config.memoryDomainFilter?.length) {
        // Remove last tag on backspace in empty input
        const tags = config.memoryDomainFilter;
        removeTag(tags[tags.length - 1]);
      }
    },
    [tagInput, addTag, removeTag, config.memoryDomainFilter],
  );

  if (loading) {
    return (
      <div className="flex items-center gap-2 py-3 px-3">
        <Loader2 className="h-3 w-3 animate-spin text-violet" />
        <span className="text-[10px] text-text-disabled">Carregando agente...</span>
      </div>
    );
  }

  return (
    <div className="space-y-3 px-3 py-3 border-t border-white/[0.04] bg-night-panel/30">
      {/* Header */}
      <div className="flex items-center gap-1.5">
        <Bot className="h-3 w-3 text-violet" />
        <span className="text-[10px] font-mono uppercase tracking-widest text-text-disabled">
          Agente
        </span>
      </div>

      {/* System Prompt */}
      <div className="space-y-1">
        <label className="text-[10px] text-text-secondary font-medium">
          System Prompt
        </label>
        <textarea
          value={config.systemPrompt ?? ""}
          onChange={(e) =>
            setConfig((prev) => ({
              ...prev,
              systemPrompt: e.target.value || null,
            }))
          }
          placeholder="Ex: Voce e um assistente especializado em financas..."
          rows={3}
          className="w-full rounded-md border border-white/[0.06] bg-deep-space/80 px-2.5 py-1.5 text-xs text-text-primary placeholder:text-text-disabled focus:border-violet/40 focus:outline-none focus:ring-1 focus:ring-violet/20 resize-y min-h-[60px]"
        />
      </div>

      {/* Memory Mode */}
      <div className="space-y-1">
        <label className="text-[10px] text-text-secondary font-medium">
          Modo de Memoria
        </label>
        <select
          value={config.memoryMode}
          onChange={(e) =>
            setConfig((prev) => ({
              ...prev,
              memoryMode: e.target.value as "global" | "filtered",
              // Clear filter when switching to global
              ...(e.target.value === "global"
                ? { memoryDomainFilter: null }
                : {}),
            }))
          }
          className="w-full rounded-md border border-white/[0.06] bg-deep-space/80 px-2.5 py-1.5 text-xs text-text-primary focus:border-violet/40 focus:outline-none focus:ring-1 focus:ring-violet/20"
        >
          <option value="global">Global (todas as memorias)</option>
          <option value="filtered">Filtrada por tags</option>
        </select>
      </div>

      {/* Tag chips — only shown when mode=filtered */}
      {config.memoryMode === "filtered" && (
        <div className="space-y-1">
          <label className="text-[10px] text-text-secondary font-medium">
            Tags de dominio
          </label>
          <div className="flex flex-wrap gap-1.5 rounded-md border border-white/[0.06] bg-deep-space/80 px-2 py-1.5 min-h-[32px] focus-within:border-violet/40 focus-within:ring-1 focus-within:ring-violet/20">
            {(config.memoryDomainFilter ?? []).map((tag) => (
              <span
                key={tag}
                className="inline-flex items-center gap-1 rounded-full bg-violet/20 px-2 py-0.5 text-[10px] font-mono text-violet"
              >
                {tag}
                <button
                  onClick={() => removeTag(tag)}
                  className="rounded-full p-0.5 hover:bg-violet/30 transition-colors"
                  aria-label={`Remover tag ${tag}`}
                >
                  <X className="h-2.5 w-2.5" />
                </button>
              </span>
            ))}
            <input
              ref={tagInputRef}
              type="text"
              value={tagInput}
              onChange={(e) => setTagInput(e.target.value)}
              onKeyDown={handleTagKeyDown}
              onBlur={() => {
                if (tagInput.trim()) addTag(tagInput);
              }}
              placeholder={
                (config.memoryDomainFilter ?? []).length === 0
                  ? "financeiro, clientes, projetos..."
                  : ""
              }
              className="flex-1 min-w-[80px] bg-transparent text-xs text-text-primary placeholder:text-text-disabled outline-none"
            />
          </div>
          <p className="text-[9px] text-text-disabled">
            Pressione Enter ou virgula para adicionar. So memorias com essas tags serao injetadas.
          </p>
        </div>
      )}

      {/* Error / Success */}
      {error && (
        <div className="flex items-center gap-1.5 text-red-400">
          <AlertCircle className="h-3 w-3" />
          <span className="text-[10px]">{error}</span>
        </div>
      )}

      {/* Save button */}
      <div className="flex items-center gap-2">
        <Button
          onClick={handleSave}
          disabled={saving}
          size="sm"
          className="h-7 px-3 text-[10px] bg-violet hover:bg-violet/80"
        >
          {saving ? (
            <Loader2 className="h-3 w-3 animate-spin mr-1" />
          ) : (
            <Save className="h-3 w-3 mr-1" />
          )}
          Salvar
        </Button>
        {success && (
          <span className="text-[10px] text-emerald-400 font-mono">Salvo!</span>
        )}
      </div>
    </div>
  );
}
```
