```typescript
export function updateTopicAgent(
  id: number,
  config: {
    systemPrompt?: string | null;
    memoryMode?: 'global' | 'filtered';
    memoryDomainFilter?: string[] | null;
  },
): boolean {
  const d = getDb();
  if (!d) return false;
  try {
    const fields: string[] = [];
    const values: (string | number | null)[] = [];

    if (config.systemPrompt !== undefined) {
      fields.push('system_prompt = ?');
      values.push(config.systemPrompt);
    }
    if (config.memoryMode !== undefined) {
      fields.push('memory_mode = ?');
      values.push(config.memoryMode);
    }
    if (config.memoryDomainFilter !== undefined) {
      fields.push('memory_domain_filter = ?');
      values.push(config.memoryDomainFilter ? JSON.stringify(config.memoryDomainFilter) : null);
    }

    if (fields.length === 0) return true;
    values.push(id);
    d.prepare(`UPDATE topics SET ${fields.join(', ')} WHERE id = ?`).run(
      ...values,
    );
    return true;
  } catch {
    return false;
  }
}
```

```typescript
export function getTopicAgent(id: number): {
  systemPrompt: string | null;
  memoryMode: 'global' | 'filtered';
  memoryDomainFilter: string[] | null;
} | null {
  const d = getDb();
  if (!d) return null;
  try {
    const row = d
      .prepare(
        "SELECT system_prompt, memory_mode, memory_domain_filter FROM topics WHERE id = ?",
      )
      .get(id) as { system_prompt: string | null; memory_mode: string; memory_domain_filter: string | null } | undefined;
    if (!row) return null;
    return {
      systemPrompt: row.system_prompt ?? null,
      memoryMode: (row.memory_mode as 'global' | 'filtered') ?? 'global',
      memoryDomainFilter: row.memory_domain_filter ? JSON.parse(row.memory_domain_filter) as string[] : null,
    };
  } catch {
    return null;
  }
}
```
