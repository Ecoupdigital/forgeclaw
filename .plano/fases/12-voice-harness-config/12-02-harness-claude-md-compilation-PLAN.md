```typescript
import { readFileSync, writeFileSync, existsSync } from 'node:fs';
import { join } from 'node:path';
import { homedir } from 'node:os';

const HARNESS_DIR = join(homedir(), '.forgeclaw', 'harness');
const OUTPUT_FILE = join(HARNESS_DIR, 'CLAUDE.md');

/**
 * Ordered list of harness files to compile into CLAUDE.md.
 * Order matters: SOUL defines identity, USER adds personal context,
 * AGENTS/TOOLS/MEMORY provide capabilities, STYLE sets output format.
 */
const HARNESS_FILES = [
  'SOUL.md',
  'USER.md',
  'AGENTS.md',
  'TOOLS.md',
  'MEMORY.md',
  'STYLE.md',
] as const;

export interface CompileHarnessResult {
  success: boolean;
  outputPath: string;
  includedFiles: string[];
  missingFiles: string[];
}

/**
 * Compile individual harness files into a single CLAUDE.md.
 *
 * Reads each file in HARNESS_FILES order, concatenates with section
 * headers, writes to ~/.forgeclaw/harness/CLAUDE.md.
 *
 * Missing files are skipped (logged in result.missingFiles).
 * At least one file must exist for compilation to succeed.
 *
 * @param harnessDir - Override harness directory (for testing). Defaults to ~/.forgeclaw/harness/
 * @returns CompileHarnessResult with success status and file lists
 */
export function compileHarness(harnessDir?: string): CompileHarnessResult {
  const dir = harnessDir ?? HARNESS_DIR;
  const outputPath = join(dir, 'CLAUDE.md');
  const includedFiles: string[] = [];
  const missingFiles: string[] = [];
  const sections: string[] = [];

  // Header
  sections.push('<!-- AUTO-GENERATED by compileHarness() — do not edit manually -->');
  sections.push('<!-- Edit individual files (SOUL.md, USER.md, etc.) and recompile -->');
  sections.push('');

  for (const fileName of HARNESS_FILES) {
    const filePath = join(dir, fileName);

    if (!existsSync(filePath)) {
      missingFiles.push(fileName);
      console.warn(`[harness-compiler] Skipping ${fileName} (not found)`);
      continue;
    }

    try {
      const content = readFileSync(filePath, 'utf-8').trim();
      if (content.length === 0) {
        missingFiles.push(fileName);
        console.warn(`[harness-compiler] Skipping ${fileName} (empty)`);
        continue;
      }

      // Section separator with file name for debugging
      sections.push(`<!-- === ${fileName} === -->`);
      sections.push(content);
      sections.push('');
      includedFiles.push(fileName);
    } catch (err) {
      missingFiles.push(fileName);
      console.error(`[harness-compiler] Error reading ${fileName}:`, err);
    }
  }

  if (includedFiles.length === 0) {
    console.error('[harness-compiler] No harness files found. CLAUDE.md not generated.');
    return { success: false, outputPath, includedFiles, missingFiles };
  }

  const compiled = sections.join('\n');
  writeFileSync(outputPath, compiled);
  console.log(`[harness-compiler] CLAUDE.md compiled from ${includedFiles.length} files (${compiled.length} bytes)`);

  return { success: true, outputPath, includedFiles, missingFiles };
}

/**
 * Check if the compiled CLAUDE.md exists and is non-empty.
 */
export function isHarnessCompiled(harnessDir?: string): boolean {
  const dir = harnessDir ?? HARNESS_DIR;
  const outputPath = join(dir, 'CLAUDE.md');

  if (!existsSync(outputPath)) return false;

  try {
    const content = readFileSync(outputPath, 'utf-8');
    return content.trim().length > 0;
  } catch {
    return false;
  }
}
```

**Decisoes de design:**
- Funcao SINCRONA (`readFileSync`/`writeFileSync`) porque e chamada durante install (sincrono) e e operacao local rapida
- Aceita `harnessDir` opcional para testes (nao depende de homedir em testes)
- Retorna objeto estruturado `CompileHarnessResult` ao inves de boolean para que o caller saiba quais files foram incluidos/faltaram
- HTML comments como separadores (nao interferem no markdown, util para debug)
- Ordem fixa e explicita dos files (SOUL primeiro define identidade, STYLE por ultimo define formato)
- Skip de arquivos faltantes com warning (install parcial ainda funciona)
- Requer pelo menos 1 arquivo para gerar output (evita CLAUDE.md vazio)

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'bun:test';
import { mkdirSync, writeFileSync, rmSync, readFileSync, existsSync } from 'node:fs';
import { join } from 'node:path';
import { tmpdir } from 'node:os';
import { compileHarness, isHarnessCompiled } from './harness-compiler';

describe('harness-compiler', () => {
  const testDir = join(tmpdir(), `harness-compiler-test-${Date.now()}`);

  beforeEach(() => {
    mkdirSync(testDir, { recursive: true });
  });

  afterEach(() => {
    rmSync(testDir, { recursive: true, force: true });
  });

  it('compiles all 6 files into CLAUDE.md', () => {
    // Create all 6 harness files
    const files = ['SOUL.md', 'USER.md', 'AGENTS.md', 'TOOLS.md', 'MEMORY.md', 'STYLE.md'];
    for (const f of files) {
      writeFileSync(join(testDir, f), `# ${f}\nContent of ${f}`);
    }

    const result = compileHarness(testDir);

    expect(result.success).toBe(true);
    expect(result.includedFiles).toEqual(files);
    expect(result.missingFiles).toEqual([]);

    const compiled = readFileSync(join(testDir, 'CLAUDE.md'), 'utf-8');
    expect(compiled).toContain('AUTO-GENERATED');
    expect(compiled).toContain('# SOUL.md');
    expect(compiled).toContain('# STYLE.md');
    // Verify order: SOUL before STYLE
    expect(compiled.indexOf('SOUL.md')).toBeLessThan(compiled.indexOf('STYLE.md'));
  });

  it('skips missing files and still succeeds', () => {
    // Only create SOUL.md and USER.md
    writeFileSync(join(testDir, 'SOUL.md'), '# Soul');
    writeFileSync(join(testDir, 'USER.md'), '# User');

    const result = compileHarness(testDir);

    expect(result.success).toBe(true);
    expect(result.includedFiles).toEqual(['SOUL.md', 'USER.md']);
    expect(result.missingFiles).toContain('AGENTS.md');
    expect(result.missingFiles).toContain('TOOLS.md');
  });

  it('fails when no harness files exist', () => {
    const result = compileHarness(testDir);

    expect(result.success).toBe(false);
    expect(result.includedFiles).toEqual([]);
    expect(existsSync(join(testDir, 'CLAUDE.md'))).toBe(false);
  });

  it('skips empty files', () => {
    writeFileSync(join(testDir, 'SOUL.md'), '# Soul content');
    writeFileSync(join(testDir, 'USER.md'), ''); // empty
    writeFileSync(join(testDir, 'AGENTS.md'), '   '); // whitespace only

    const result = compileHarness(testDir);

    expect(result.success).toBe(true);
    expect(result.includedFiles).toEqual(['SOUL.md']);
    expect(result.missingFiles).toContain('USER.md');
    expect(result.missingFiles).toContain('AGENTS.md');
  });

  describe('isHarnessCompiled', () => {
    it('returns false when CLAUDE.md does not exist', () => {
      expect(isHarnessCompiled(testDir)).toBe(false);
    });

    it('returns false when CLAUDE.md is empty', () => {
      writeFileSync(join(testDir, 'CLAUDE.md'), '');
      expect(isHarnessCompiled(testDir)).toBe(false);
    });

    it('returns true when CLAUDE.md has content', () => {
      writeFileSync(join(testDir, 'CLAUDE.md'), '# Compiled harness');
      expect(isHarnessCompiled(testDir)).toBe(true);
    });
  });
});
```

**Notas:**
- Usa tmpdir para isolamento total (nao toca em ~/.forgeclaw real)
- Limpa diretorio temporario apos cada teste
- Testa: compilacao completa, files parciais, nenhum file, files vazios, isHarnessCompiled
